            }

        }


        function addMessage(text, sender) {

            const messagesDiv = document.getElementById('chatMessages');

            const messageDiv = document.createElement('div');

            messageDiv.className = `message ${sender}`;

            messageDiv.innerHTML = `<p>${text}</p>`;

            messagesDiv.appendChild(messageDiv);

            messagesDiv.scrollTop = messagesDiv.scrollHeight;

        }


        function generateAIResponse(userMessage) {

            const responses = {

                'career': "I can help you explore career opportunities in Saudi Arabia! What field interests you most? Technology, healthcare, energy, or tourism?",

                'skills': "Great question! Based on Saudi Vision 2030, key skills include digital literacy, critical thinking, communication, and adaptability. What's your current skill level?",

                'courses': "I'd recommend starting with courses in your field of interest. Would you like suggestions for technology, business, or creative skills?",

                'cv': "I can help you improve your CV! Make sure to highlight relevant experience, use action verbs, and tailor it to Saudi job market expectations.",

                'default': "That's an interesting question! I'm here to help with career guidance, skill development, and course recommendations. How can I assist you with your professional journey?"

            };


            const lowerMessage = userMessage.toLowerCase();

            

            if (lowerMessage.includes('career') || lowerMessage.includes('job')) {

                return responses.career;

            } else if (lowerMessage.includes('skill') || lowerMessage.includes('learn')) {

                return responses.skills;

            } else if (lowerMessage.includes('course') || lowerMessage.includes('training')) {

                return responses.courses;

            } else if (lowerMessage.includes('cv') || lowerMessage.includes('resume')) {

                return responses.cv;

            } else {

                return responses.default;

            }

        }

        function selectOption(element, category, value) {

            // Remove selection from other options in the same category

            const options = element.parentNode.querySelectorAll('.option');

            options.forEach(opt => opt.classList.remove('selected'));

            

            // Select this option

            assessmentAnswers[category] = value;

        }
